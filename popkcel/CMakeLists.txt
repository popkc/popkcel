if(WIN32)
    set(ELPS win.cpp)
    add_definitions(-DPOPKCELNOLOCK)
else(WIN32)
    set(ELPS unix.cpp)
    if(${CMAKE_SYSTEM_NAME} STREQUAL "Linux")
        list(APPEND ELPS linux.cpp)
    else(${CMAKE_SYSTEM_NAME} STREQUAL "Linux")
        list(APPEND ELPS bsd.cpp)
    endif(${CMAKE_SYSTEM_NAME} STREQUAL "Linux")
endif(WIN32)

if(MSVC)
	enable_language(ASM_MASM)
	if(CMAKE_SIZEOF_VOID_P EQUAL 4)
        list(APPEND ELPS winx86.asm)
        set_source_files_properties(winx86.asm PROPERTIES COMPILE_FLAGS "/safeseh")
	else(CMAKE_SIZEOF_VOID_P EQUAL 4)
		list(APPEND ELPS winx64.asm)
	endif(CMAKE_SIZEOF_VOID_P EQUAL 4)
endif(MSVC)

add_library(popkcel ${ELBT} popkcel.cpp popkcel.h ${ELPS})
add_definitions(-D_FORTIFY_SOURCE=0)

if(UNIX OR MINGW)
target_link_libraries(popkcel PUBLIC pthread)
endif(UNIX OR MINGW)

if(WIN32)
target_link_libraries(popkcel PUBLIC WS2_32)
endif(WIN32)
